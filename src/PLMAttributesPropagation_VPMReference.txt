	Parameters	RuleContext

Let sPropagationMode, sEditAttributeId(String)
Let pBusinessType(ValuePointer)
Let lBusinessType(List)

Trace(1, "==> Attributes valuation propagation - VPMReference")
Trace(2, "ThisObject Name: #", ThisObject.Name)

sPropagationMode = Parameters.GetAttributeString("PropagationMode")
sEditAttributeId = Parameters.GetAttributeString("EditAttributeId")
Trace(2, "Propagation Mode: #", sPropagationMode)
Trace(2, "Edit Attribute Id: #", sEditAttributeId)

// Configure Business Logic for VPMReference objects to automatically set the VversionComment attribute to 'Test Value'.
// This logic sets the default value only if VversionComment is currently empty, allowing user edits to persist.
if (ThisObject.VversionComment.IsEmpty()) {
    ThisObject.SetAttributeString("VversionComment", "Test Value")
    Trace(2, "VversionComment was empty, set to default: 'Test Value'")
}

if(sPropagationMode == "Edition")
{
	ThisObject.SetAttributeString("V_Name", "PLM_ExternalID")

    // Upon user edit of VversionComment, enforce a minimum character limit of 20.
    if (sEditAttributeId == "VversionComment") {
        Let sCurrentComment(String)
        sCurrentComment = ThisObject.VversionComment // Get the value currently being edited by the user

        Trace(2, "User is editing VversionComment. Current value: '", sCurrentComment, "'")
        Trace(2, "Length of current VversionComment: ", sCurrentComment.Length())

        // If the input is less than 20 characters, stop execution and prompt the user with an error.
        if (sCurrentComment.Length() < 20) {
            Message("VversionComment must be at least 20 characters long. Current length is " + sCurrentComment.Length() + ". Please modify the comment.")
            // The Message keyword will typically stop the transaction and display the error to the user.
        } else {
            Trace(2, "VversionComment meets the minimum length requirement (>= 20 characters).")
        }
    }
}

Trace(1, "Attributes valuation propagation ==>")
