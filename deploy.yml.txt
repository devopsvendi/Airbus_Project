---
- name: Deploy Gradle ZIP Package via Ansible
  hosts: localhost
  vars:
    zip_path: "C:/ProgramData/Jenkins/.jenkins/workspace/Gradle_project/build/distributions/Gradle_project.zip"
    deploy_to: "C:/EKL-Project/production-deployment"
  
  tasks:
    - name: Ensure target deployment directory exists
      ansible.windows.win_file:
        path: "{{ deploy_to }}"
        state: directory

    - name: Unzip deployment package to target environment
      ansible.windows.win_unzip:
        src: "{{ zip_path }}"
        dest: "{{ deploy_to }}"
        rm: false
